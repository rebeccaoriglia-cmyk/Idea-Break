const personaggi = [

const luoghi = [

  "un sentiero di montagna al mattino",
  "un garage pieno di attrezzi",
  "una stanza illuminata solo dal PC",
  "una spiaggia fuori stagione",
  "una piazzetta di un paesino"

const oggetti = [
  "un elastico",
  "una tazza mezza piena",
  "un bullone",
  "una conchiglia",
  "un foglio spiegazzato",
  "un vecchio biglietto"
];

const output = document.getElementById("output");

// Event listener dei pulsanti
document.querySelectorAll("button").forEach(btn => {
  btn.addEventListener("click", () => {
    const category = btn.dataset.category;

    // --- Se è la SCENA RAPIDA ---
    if (category === "scena") {
      const p = personaggi[Math.floor(Math.random() * personaggi.length)];
      const l = luoghi[Math.floor(Math.random() * luoghi.length)];
      const o = oggetti[Math.floor(Math.random() * oggetti.length)];

      const scenaPrompt =
        `Personaggio: ${p}\nLuogo: ${l}\nOggetto: ${o}\n\n` +
        "Scrivi una micro-scena (max 1 riga) usando questi tre elementi.";

      output.textContent = scenaPrompt;
      output.classList.add("visible");
      return;
    }

    // --- Altrimenti categorie normali ---
    const list = prompts[category];
    const randomPrompt = list[Math.floor(Math.random() * list.length)];

    output.textContent = randomPrompt;
    output.classList.add("visible");
  });
});
